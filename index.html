fetch("voters.json") .then(r => r.json()) .then(data => { voterData = data; }); function searchData(){ let q = document.getElementById("searchBox") .value .toLowerCase(); let html = ""; let count = 0; voterData.forEach(v => { if( JSON.stringify(v) .toLowerCase() .includes(q) ){ count++; html += `
Name/Details: ${v.details}
House No: ${v.house_no}
Age: ${v.age}
EPIC: ${v.epic}
PDF: ${v.pdf}
`; } }); document.getElementById("results").innerHTML = `

${count} matches

` + html; }
